    ```go
    package selfupdate

    import (
        "context"
        "fmt"
        "log/slog"
        "sync/atomic"

        "github.com/centroid-is/docker-update/internal/docker"
        "github.com/moby/moby/api/types/container"
    )

    type Spawner interface {
        Spawn(ctx context.Context) (helperID string, err error)
    }

    type spawner struct {
        cli               docker.Client
        selfImage         string  // e.g. "ghcr.io/centroid-is/docker-update:<digest>"
        selfContainer     string  // e.g. "docker-update" (the watched-service name)
        inFlight          atomic.Bool
        actionsInFlightFn func() int  // returns count of in-flight per-service actions; 0 means safe
        helperKeepAlive   bool  // from DOCKER_UPDATE_SELF_UPDATE_KEEP_HELPER env, default false
    }

    func NewSpawner(cli docker.Client, selfImage, selfContainer string, actionsInFlightFn func() int, keepHelper bool) Spawner {
        return &spawner{
            cli: cli,
            selfImage: selfImage,
            selfContainer: selfContainer,
            actionsInFlightFn: actionsInFlightFn,
            helperKeepAlive: keepHelper,
        }
    }

    var (
        ErrSelfUpdateInFlight = fmt.Errorf("self_update_in_flight")
        ErrActionsInFlight    = fmt.Errorf("actions_in_flight")
    )

    func (s *spawner) Spawn(ctx context.Context) (string, error) {
        // Open Question 5: refuse if any per-service action is running
        if s.actionsInFlightFn != nil && s.actionsInFlightFn() > 0 {
            return "", ErrActionsInFlight
        }
        // Open Question 2: refuse if a self-update is already in flight
        if !s.inFlight.CompareAndSwap(false, true) {
            return "", ErrSelfUpdateInFlight
        }
        // do NOT defer reset — inFlight stays true until the helper SIGTERM's us out

        // Build helper ContainerCreateOptions per RESEARCH.md Pattern 4:
        opts := docker.ContainerCreateOptions{
            Config: &container.Config{
                Image: s.selfImage,
                Cmd:   []string{"docker-update", "--self-update-orchestrator", "--target=" + s.selfContainer},
                Labels: map[string]string{
                    "centroid.docker-update.helper": "true",
                },
            },
            HostConfig: &container.HostConfig{
                AutoRemove: !s.helperKeepAlive,  // default true; opt-out via KEEP_HELPER env
                Binds: []string{
                    "/var/run/docker.sock:/var/run/docker.sock",
                },
            },
            Name: "", // unnamed; let the daemon assign — auto-remove cleans up
        }

        res, err := s.cli.ContainerCreate(ctx, opts)
        if err != nil {
            s.inFlight.Store(false)  // reset on failure so retry is possible
            return "", fmt.Errorf("selfupdate.Spawn: create helper: %w", err)
        }

        if err := s.cli.ContainerStart(ctx, res.ID, docker.ContainerStartOptions{}); err != nil {
            // best-effort cleanup of the just-created helper
            _ = s.cli.ContainerRemove(ctx, res.ID, docker.ContainerRemoveOptions{Force: true})
            s.inFlight.Store(false)
            return "", fmt.Errorf("selfupdate.Spawn: start helper %s: %w", res.ID, err)
        }

        slog.Info("self_update.helper_spawned", "helper_id", res.ID, "target", s.selfContainer)
        return res.ID, nil
    }
    ```

    Notes:
      - The Spawner is per-process state (inFlight is a process-local atomic). Acceptable per Open Question 2 — operator clicking twice in quick succession hits the same process.
      - actionsInFlightFn comes from the orchestrator's mutex-map cardinality (wired in main.go in Task 3).
      - `selfImage` is set at parent boot time.     package selfupdate

    ```go
    import (
        "context"
        "fmt"
        "log/slog"
        "net/http"
        "time"

        "github.com/centroid-is/docker-update/internal/docker"
        "github.com/centroid-is/docker-update/internal/recreate"
    )

    // Orchestrate is the helper-side entry point. Called by main.go when
    // --self-update-orchestrator is set.
    //
    // Sequence per RESEARCH.md Pattern 4:
    //   1. Wait `delay` (lets parent return 202)
    //   2. Call recreate.Service(ctx, cli, target) — same primitive as Plan 09-03
    //   3. Poll healthzURL until 200 OR `verifyTimeout` elapses
    //   4. Exit 0 on success; non-zero on failure (helper logs preserved if KEEP_HELPER=true)
    func Orchestrate(
        ctx context.Context,
        cli docker.Client,
        target string,
        healthzURL string,
        delay time.Duration,
        verifyTimeout time.Duration,
    ) error {
        slog.Info("self_update.orchestrate.start", "target", target, "delay", delay, "verify_timeout", verifyTimeout)
        time.Sleep(delay)

        newID, err := recreate.Service(ctx, cli, target)
        if err != nil {
            slog.Error("self_update.orchestrate.recreate_failed", "err", err)
            return fmt.Errorf("orchestrate: recreate %s: %w", target, err)
        }
        slog.Info("self_update.orchestrate.recreate_done", "new_id", newID)

        // Poll healthzURL
        deadline := time.Now().Add(verifyTimeout)
        tick := time.NewTicker(2 * time.Second)
        defer tick.Stop()
        client := &http.Client{Timeout: 3 * time.Second}

        for time.Now().Before(deadline) {
            resp, err := client.Get(healthzURL)
            if err == nil && resp.StatusCode == 200 {
                _ = resp.Body.Close()
                slog.Info("self_update.orchestrate.verify_ok", "new_id", newID)
                return nil
            }
            if resp != nil { _ = resp.Body.Close() }
            select {
            case <-ctx.Done():
                return ctx.Err()
            case <-tick.C:
            }
        }
        slog.Error("self_update.orchestrate.verify_timeout", "new_id", newID, "timeout", verifyTimeout)
        return fmt.Errorf("orchestrate: verify timeout after %v", verifyTimeout)
    }
    ```

    Notes:
      - `healthzURL` is built in main.go from the target's container IP or a known port mapping. For v1, pass it explicitly as `http://<container_ip>:8080/healthz` (resolved from `ContainerInspect.NetworkSettings.Networks[*].IPAddress`).
      - The 2-second poll tick is hard-coded (60s timeout / 2s tick = ~30 attempts; matches RESEARCH.md's "60s is 20x safety margin").

    ```go
    package api

    import (
        "context"
        "errors"
        "log/slog"
        "net/http"

        "github.com/centroid-is/docker-update/internal/selfupdate"
    )

    const (
        actionBodySelfUpdaterUnwired = `{"error":"self_updater_not_wired","detail":"restart docker-update; check boot logs"}`
        actionBodySelfUpdateFailed   = `{"error":"self_update_failed","detail":"see logs for self_update.spawn_failed event"}`
        actionBodyActionsInFlight    = `{"error":"actions_in_flight","detail":"wait for in-flight actions to complete"}`
        actionBodySelfUpdateBusy     = `{"error":"self_update_in_flight","detail":"a self-update is already running"}`
    )

    func (s *Server) handleSelfUpdate(w http.ResponseWriter, r *http.Request) {
        if s.selfUpdater == nil {
            writeActionBody(w, http.StatusServiceUnavailable, actionBodySelfUpdaterUnwired)
            return
        }
        helperID, err := s.selfUpdater.Spawn(r.Context())
        if err != nil {
            // Sentinel mapping (per RESEARCH.md Open Questions 2 and 5)
            switch {
            case errors.Is(err, selfupdate.ErrActionsInFlight):
                writeActionBody(w, http.StatusConflict, actionBodyActionsInFlight)
                return
            case errors.Is(err, selfupdate.ErrSelfUpdateInFlight):
                writeActionBody(w, http.StatusConflict, actionBodySelfUpdateBusy)
                return
            }
            slog.Error("self_update.spawn_failed", "err", err)
    ```
