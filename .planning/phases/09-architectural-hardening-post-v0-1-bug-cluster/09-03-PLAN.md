    ```go
    package recreate

    import (
        "context"
        "fmt"
        "strings"

        "github.com/centroid-is/docker-update/internal/docker"
    )

    // Service performs Stop → Remove → Create → NetworkConnect (extras) → Start.
    // Returns new container ID on success.
    //
    // Single-service contract (per RESEARCH.md Open Question 3 RESOLVED): this
    // function recreates exactly ONE service per call; multi-service coordination
    // is explicitly out of scope for Phase 9.
    //
    // Container lookup mirrors the canonical analog
    // internal/actions/orchestrator.go::lookupContainerIDByService — same
    // facade types (docker.ContainerListOptions, docker.Filters), same compose
    // label format, same All=true rationale. Use that function as the
    // reference pattern; the only divergence here is that recreate.Service
    // returns the new container ID rather than the looked-up old one.
    func Service(ctx context.Context, cli docker.Client, svcName string) (string, error) {
        // 1. lookup container by compose service label (mirror of
        //    actions/orchestrator.go::lookupContainerIDByService — same facade
        //    types from internal/docker/client.go; B1 fix: there is NO
        //    container.ListOptions type in the moby SDK, the facade alias is
        //    docker.ContainerListOptions = client.ContainerListOptions and
        //    docker.Filters = client.Filters).
        list, err := cli.ContainerList(ctx, docker.ContainerListOptions{
            All: true,
            Filters: docker.Filters{
                "label": {"com.docker.compose.service=" + svcName: true},
            },
        })
        if err != nil { return "", fmt.Errorf("recreate.Service: list %q: %w", svcName, err) }
        if len(list) == 0 { return "", fmt.Errorf("recreate.Service: no container for service %q", svcName) }
        // Most-recently-created wins (defensive against an in-flight prior
        // recreate that left a dying container behind) — same heuristic as
        // orchestrator.go::lookupContainerIDByService: pick max(Created)
        // explicitly because the moby SDK does not guarantee ordering on
        // the wire. Copy that function's tie-break (stable first-occurrence
        // on the SDK-returned slice).
        newest := list[0]
        for _, c := range list[1:] {
            if c.Created > newest.Created {
                newest = c
            }
        }
        oldID := newest.ID

        // 2. inspect
        inspect, err := cli.ContainerInspect(ctx, oldID)
        if err != nil { return "", fmt.Errorf("recreate.Service: inspect %s: %w", oldID, err) }

        // 3. translate
        cfg, hostCfg, netCfg, extraNets := Translate(inspect)
        name := strings.TrimPrefix(inspect.Name, "/")

        // 4. Stop (10s grace per compose default)
        stopTimeout := 10
        if err := cli.ContainerStop(ctx, oldID, docker.ContainerStopOptions{Timeout: &stopTimeout}); err != nil {
            return "", fmt.Errorf("recreate.Service: stop %s: %w", oldID, err)
        }

        // 5. Remove (Force=true so attached/stopped containers do not block; RemoveVolumes=false)
        if err := cli.ContainerRemove(ctx, oldID, docker.ContainerRemoveOptions{Force: true}); err != nil {
            return "", fmt.Errorf("recreate.Service: remove %s: %w", oldID, err)
        }

        // 6. Create (CRITICAL: old container is now GONE — if this fails, return loudly)
        res, err := cli.ContainerCreate(ctx, docker.ContainerCreateOptions{
            Config: cfg, HostConfig: hostCfg, NetworkingConfig: netCfg, Name: name,
        })
        if err != nil {
            return "", fmt.Errorf("recreate.Service: create %s (old GONE): %w", name, err)
        }

        // 7. Connect extra networks (best-effort cleanup of new container on failure)
        for netName, eps := range extraNets {
            if err := cli.NetworkConnect(ctx, netName, docker.NetworkConnectOptions{Container: res.ID, EndpointSettings: eps}); err != nil {
                _ = cli.ContainerRemove(ctx, res.ID, docker.ContainerRemoveOptions{Force: true})
                return "", fmt.Errorf("recreate.Service: connect %s to %s: %w", res.ID, netName, err)
            }
        }

        // 8. Start (best-effort cleanup on failure)
        if err := cli.ContainerStart(ctx, res.ID, docker.ContainerStartOptions{}); err != nil {
            _ = cli.ContainerRemove(ctx, res.ID, docker.ContainerRemoveOptions{Force: true})
            return "", fmt.Errorf("recreate.Service: start %s: %w", res.ID, err)
        }
        return res.ID, nil
    }
    ```
