    //
    // Pushes an OCI image index with two children (linux/amd64 + linux/arm64),
    // then returns the AMD64 CHILD manifest's digest. crane.Digest(WithPlatform(amd64))
    // against this index should return the same digest — Acceptance criterion
    // for DETECT-01 + DETECT-04 (multi-arch index resolution).
    //
    // Why crane CLI and not oras: oras does not have first-class OCI image
    // index construction. crane's `index append` is the canonical tool —
    // it's the same Go library (go-containerregistry) the resolver wraps.
    //
    // crane is expected on PATH. CI's workflow installs it; local dev: see
    // README onboarding.

    import { execSync } from 'node:child_process';

    const ZOT_HOST_PORT = process.env.ZOT_HOST_PORT ?? '15000';

    export function pushFreshIndex(repo: string): string {
      const stamp = `${Date.now()}-${Math.random().toString(36).slice(2)}`;
      const amd64Ref = `localhost:${ZOT_HOST_PORT}/${repo}:amd64-${stamp}`;
      const arm64Ref = `localhost:${ZOT_HOST_PORT}/${repo}:arm64-${stamp}`;
      const indexRef = `localhost:${ZOT_HOST_PORT}/${repo}:latest`;

      // Build two single-arch placeholder children. crane.append creates a
      // 2-layer image (config + one tar) per child. The content doesn't
      // matter — the digest does.
      const amd64Tar = `/tmp/amd64-${stamp}.tar.gz`;
      const arm64Tar = `/tmp/arm64-${stamp}.tar.gz`;
      execSync(`tar -czf ${amd64Tar} -T /dev/null`, { stdio: 'pipe' });
      execSync(`tar -czf ${arm64Tar} -T /dev/null`, { stdio: 'pipe' });

      execSync(
        `crane append --insecure --new_tag ${amd64Ref} --platform linux/amd64 -f ${amd64Tar}`,
        { stdio: 'pipe' },
      );
      execSync(
        `crane append --insecure --new_tag ${arm64Ref} --platform linux/arm64 -f ${arm64Tar}`,
        { stdio: 'pipe' },
      );

      // Get the AMD64 child digest BEFORE building the index — that's what
      // the resolver should return.
      const amd64Digest = execSync(
        `crane digest --insecure --platform linux/amd64 ${amd64Ref}`,
        { encoding: 'utf8' },
      ).trim();

      // Build and push the OCI index pointing at both children, tagged :latest.
      execSync(
        `crane index append --insecure --tag ${indexRef} -m ${amd64Ref} -m ${arm64Ref}`,
        { stdio: 'pipe' },
      );

      if (!/^sha256:[0-9a-f]+$/.test(amd64Digest)) {
        throw new Error(`crane returned an invalid amd64 digest: ${amd64Digest}`);
      }
      return amd64Digest;
    }
