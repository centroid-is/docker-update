    Body per RESEARCH.md lines 568–591:

    ```go
    func (o *actionOrchestrator) lockService(svc string) (func(), error) {
        o.mu.RLock()
        m, ok := o.locks[svc]
        o.mu.RUnlock()
        if !ok {
            o.mu.Lock()
            m, ok = o.locks[svc]
            if !ok {
                m = &sync.Mutex{}
                o.locks[svc] = m
            }
            o.mu.Unlock()
        }
        if !m.TryLock() {
            return nil, ErrServiceBusy
        }
        return m.Unlock, nil
    }
    ```

    File header (Pattern A) explains why double-checked locking is necessary (two concurrent first-time requests for the same svc could otherwise create two distinct mutex instances).
