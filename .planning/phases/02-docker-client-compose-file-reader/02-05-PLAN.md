       ```yaml
       # ============================================================================
       # compose.test.override.eacces.yml
       #
       # PURPOSE
       #   Forces the hmi-update service to run with a GID that does not match the
       #   host's docker group, so the bind-mounted /var/run/docker.sock returns
       #   EACCES on connect(). This drives /healthz into the
       #   `healthzBodySocketEACCES` branch and lets the Playwright
       #   healthz-negative spec assert the Pitfall 9 remediation hint string.
       #
       #   See research/PITFALLS.md Pitfall 9 for the canonical failure mode and
       #   .planning/phases/02-docker-client-compose-file-reader/02-CONTEXT.md
       #   "Healthz Remediation Hints (DOCK-03)" for the verbatim hint string.
       #
       # USAGE
       #   docker compose -f e2e/compose.test.yml -f e2e/compose.test.override.eacces.yml up -d
       #
       # MERGE BEHAVIOUR
       #   Compose v2 merges service blocks key-by-key for scalar fields (user,
       #   image, command, etc.). For ARRAY fields like `volumes` and `environment`
       #   compose APPENDS by default in v2; we rely on that here — the override
       #   does not touch volumes, so the base compose.test.yml's
       #   /var/run/docker.sock bind-mount is preserved AS-IS. Only the `user`
       #   field is overridden.
       #
       #   We pin user="65532:65532" deliberately: UID 65532 (nonroot in
       #   distroless) and GID 65532 are guaranteed NOT to match the host's
       #   docker group GID (which is 998/999/etc. on every realistic Debian or
       #   GitHub Actions runner). The socket bind-mount remains readable in
       #   terms of stat (mode 0o660, owner root:docker) but connect() refuses
       #   because the container user is not in the docker group.
       # ============================================================================

       services:
         hmi-update:
           # Override only the user field. Everything else (volumes, ports,
           # environment, depends_on, tmpfs) inherits from compose.test.yml.
           user: "65532:65532"
       ```

    2. Create `e2e/compose.test.override.no-socket.yml`. Same header convention; this override is trickier because compose's array-merge for `volumes:` will APPEND the override's list, which would not remove anything. To get the socket bind-mount truly off, we rebind `volumes:` from scratch via a comment + a workaround: the cleanest approach is to set the docker socket path env to a deliberately-missing path so the stat fails:
       ```yaml
       # ============================================================================
       # compose.test.override.no-socket.yml
       #
       # PURPOSE
       #   Drives /healthz into the `healthzBodySocketMissing` branch by pointing
       #   HMI_UPDATE_DOCKER_HOST at a path that does not exist inside the
       #   container. This is more robust than trying to remove the socket
       #   bind-mount from compose.test.yml — compose v2 APPENDS volumes from
       #   overrides; there is no clean way to *delete* a specific volume entry,
       #   only to add more.
       #
       #   The /healthz handler reads dockerSocketPath() (in internal/api/handlers.go)
       #   which honours HMI_UPDATE_DOCKER_HOST. Pointing it at
       #   /var/run/does-not-exist.sock makes os.Stat return fs.ErrNotExist and
       #   the handler emits the "docker socket missing" body verbatim.
       #
       # USAGE
       #   docker compose -f e2e/compose.test.yml -f e2e/compose.test.override.no-socket.yml up -d
       #
       # NOTE
       #   This means the socket IS still bind-mounted; the test only proves the
       #   "operator did not configure the bind-mount" branch of /healthz, not
       #   the actual missing-mount filesystem case. For Phase 2 acceptance the
       #   handler-level branch coverage is what matters; the operator's
       #   real-world failure (missing bind-mount) is the same code path.
       # ============================================================================

       services:
         hmi-update:
           environment:
             - HMI_UPDATE_DOCKER_HOST=/var/run/does-not-exist.sock
       ```

       This is a calibrated compromise: a "true" no-socket override would require touching the base compose.test.yml. The env-var redirect proves the handler branch which is what DOCK-03 / OBS-02 require. Document the compromise in the file's comment — never leave that decision implicit.

       // DOCK-04 — Containers labeled hmi-update.watch=true are visible in

         service: string,

       ```typescript
       // OBS-02 / DOCK-03 — /healthz negative-path coverage. Each test
       // tears down the base stack, brings up the override stack, runs its
       // assertions, then tears down. The afterAll restores the base stack
       // so subsequent specs (and the global teardown) see a consistent
       // baseline.
       //
       // This is ~30s of overhead — acceptable for what it proves.

       import { execSync } from 'node:child_process';
       import { setTimeout as sleep } from 'node:timers/promises';
       import { expect, test } from '@playwright/test';

       const COMPOSE_BASE = 'docker compose -f compose.test.yml';

       async function waitForHealth(url: string, status: number, timeoutMs: number): Promise<Response> {
         const deadline = Date.now() + timeoutMs;
         let last: Response | null = null;
         while (Date.now() < deadline) {
           try {
             const r = await fetch(url);
             last = r;
             if (r.status === status) return r;
           } catch { /* network not up yet */ }
           await sleep(500);
         }
         if (last) return last;
         throw new Error(`never reached ${url}`);
       }

       async function downAndUpWithOverride(override: string) {
         execSync(`${COMPOSE_BASE} down -v --remove-orphans`, { stdio: 'pipe' });
         execSync(`${COMPOSE_BASE} -f ${override} up -d --wait`, { stdio: 'inherit' });
       }

       async function restoreBaseStack() {
         execSync(`${COMPOSE_BASE} down -v --remove-orphans`, { stdio: 'pipe' });
         execSync(`${COMPOSE_BASE} up -d --wait`, { stdio: 'inherit' });
         // Re-run the global-setup-equivalent zot wait so subsequent tests
         // (if any in same run) get a healthy stack.
         await waitForHealth('http://localhost:15000/v2/', 200, 30_000);
       }

       test.describe.serial('healthz negative-path coverage', () => {
         test.afterAll(async () => {
           await restoreBaseStack();
         });

         test('eacces: /healthz returns 503 with the Pitfall 9 remediation hint', async () => {
           await downAndUpWithOverride('compose.test.override.eacces.yml');
           const r = await waitForHealth('http://localhost:8080/healthz', 503, 30_000);
           expect(r.status).toBe(503);
           const body = await r.json();
           expect(body.status).toBe('unhealthy');
           // Verbatim Pitfall 9 hint per CONTEXT.md "Healthz Remediation Hints"

The simplest shape: probe the endpoint first; if it returns 404, skip the test with a clear message. Otherwise atomic-rename and assert.

       ```typescript
       // DOCK-02 — Compose file replaced via atomic rename triggers
       // GET /debug/compose-stat returning 412 + the verbatim drift response.
       //
       // The /debug/compose-stat route is registered only when the binary is
       // built with -tags=debug. This spec auto-skips when the route is not
       // present (production build) — that lets it co-exist with the regular
       // `make e2e` run without breaking the production-image path.

       import { execSync } from 'node:child_process';
       import { readFileSync, renameSync, writeFileSync } from 'node:fs';
       import { setTimeout as sleep } from 'node:timers/promises';
       import { expect, test } from '@playwright/test';

       test('compose drift: atomic rename of compose file detected by /debug/compose-stat', async ({ request }) => {
         // Precondition: skip when the debug route is not registered. This
         // distinguishes a production image (404) from a debug image
         // (200 ok / 412 moved).
         const probe = await request.get('/debug/compose-stat');
         if (probe.status() === 404) {
           test.skip(true, '/debug/compose-stat not registered — production build (run via `make e2e-debug` for this spec)');
         }

         // The hmi-update container bind-mounts e2e/compose.test.yml to
         // /host/docker-compose.yml. The READER inside the container stat'd
         // the IN-CONTAINER path at boot. Modifying the HOST file via
         // atomic rename changes the bind-mounted file's inode from the
         // container's perspective too.
         const composePath = './compose.test.yml';
         const original = readFileSync(composePath);
         const tmp = `${composePath}.drift-${Date.now()}`;
         writeFileSync(tmp, original); // identical content; only the inode changes
         try {
           renameSync(tmp, composePath); // atomic; inode flips
           // Give the container a beat to see the new inode (no async path
           // needed — stat is synchronous from the container's POV).
           await sleep(200);

           const drift = await request.get('/debug/compose-stat');
           expect(drift.status()).toBe(412);
           const body = await drift.json();
           expect(body.error).toBe('compose_file_moved');
           expect(body.hint).toBe('restart hmi-update to pick up the new docker-compose.yml');
         } finally {
           // Best-effort restore: write the original content back so other
           // tests that re-read the compose file see consistent contents.
           // (Inode drift remains — the reader will still error until the
           // process restarts; that's the intended behaviour. The next
           // global teardown brings the stack down so it doesn't matter.)
           try { writeFileSync(composePath, original); } catch { /* ignore */ }
         }
       });
       ```
