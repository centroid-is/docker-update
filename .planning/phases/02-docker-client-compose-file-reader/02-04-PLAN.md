       ```go
       package api

       import (
           "context"
           "encoding/json"
           "errors"
           "io/fs"
           "log/slog"
           "net/http"
           "os"
           "strings"
           "syscall"
           "time"

           "github.com/centroid-is/hmi-update/internal/state"
       )

       // dockerSocketPath returns the path that healthz stats to determine
       // socket reachability. Overridable via HMI_UPDATE_DOCKER_HOST for
       // tests (per CONTEXT.md "Healthz Remediation Hints" step 1).
       func dockerSocketPath() string {
           if v := os.Getenv("HMI_UPDATE_DOCKER_HOST"); v != "" {
               return v
           }
           return "/var/run/docker.sock"
       }

       const (
           healthzBodyOK              = `{"status":"ok"}`
           healthzBodySocketEACCES    = `{"status":"unhealthy","reason":"docker socket permission denied — set compose user: '65532:$(id -g docker)' (Pitfall 9)"}`
           healthzBodySocketMissing   = `{"status":"unhealthy","reason":"docker socket missing — add bind-mount '/var/run/docker.sock:/var/run/docker.sock'"}`
           healthzBodyDaemonUnreach   = `{"status":"unhealthy","reason":"docker daemon unreachable"}`
           healthzBodyStateUnwired    = `{"status":"unhealthy","reason":"state store unavailable; check HMI_UPDATE_STATE_PATH and restart"}`
       )

       // healthz returns 200 only when state store, docker socket, and
       // docker daemon Ping are all green. 503 on any failure with a
       // remediation hint per CONTEXT.md "Healthz Remediation Hints (DOCK-03)".
       //
       // Detection flow (CONTEXT.md step 1-3):
       //   1. State store wired?            -> nil store: 503 healthzBodyStateUnwired
       //   2. os.Stat(dockerSocketPath()):
       //        fs.ErrNotExist    -> 503 healthzBodySocketMissing
       //        fs.ErrPermission  -> 503 healthzBodySocketEACCES
       //   3. dockerClient.Ping(ctx500ms):
       //        syscall.EACCES (or "permission denied" string match) -> 503 healthzBodySocketEACCES
       //        any other error                                       -> 503 healthzBodyDaemonUnreach
       //   4. All green -> 200 healthzBodyOK
       //
       // Security invariant (T-01-04-03): the response body MUST NOT echo
       // any absolute path from os.Stat / process state. The verbatim hint
       // strings reference '/var/run/docker.sock' which is a documented
       // user-advice string (Pitfall 9), not a process-state leak.
       func (s *Server) healthz(w http.ResponseWriter, r *http.Request) {
           w.Header().Set("Content-Type", "application/json; charset=utf-8")
           w.Header().Set("Cache-Control", "no-store")

           if s.store == nil {
               w.WriteHeader(http.StatusServiceUnavailable)
               _, _ = w.Write([]byte(healthzBodyStateUnwired))
               return
           }
           // Touch the store to prove it's still readable. Phase 1 contract.
           _ = s.store.Get()

           // If dockerClient is nil we treat as socket-missing — this is a
           // wiring bug. Tests cover the explicit branches; this is a
           // belt-and-braces guard.
           if s.dockerClient == nil {
               w.WriteHeader(http.StatusServiceUnavailable)
               _, _ = w.Write([]byte(healthzBodySocketMissing))
               return
           }

           // Step 2: stat the socket.
           if _, err := os.Stat(dockerSocketPath()); err != nil {
               switch {
               case errors.Is(err, fs.ErrNotExist):
                   w.WriteHeader(http.StatusServiceUnavailable)
                   _, _ = w.Write([]byte(healthzBodySocketMissing))
                   return
               case errors.Is(err, fs.ErrPermission):
                   w.WriteHeader(http.StatusServiceUnavailable)
                   _, _ = w.Write([]byte(healthzBodySocketEACCES))
                   return
               default:
                   // Other stat errors are surfaced as "daemon unreachable"
                   // — we still don't echo the path or the underlying
                   // sys-call error string.
                   slog.Warn("healthz.socket.stat.fail", "err", err)
                   w.WriteHeader(http.StatusServiceUnavailable)
                   _, _ = w.Write([]byte(healthzBodyDaemonUnreach))
                   return
               }
           }

           // Step 3: Ping with a 500ms timeout (CONTEXT.md "Claude's
           // Discretion" — 500ms preferred, fails fast under wedge).
           pingCtx, cancel := context.WithTimeout(r.Context(), 500*time.Millisecond)
           defer cancel()
           if err := s.dockerClient.Ping(pingCtx); err != nil {
               if errors.Is(err, syscall.EACCES) || strings.Contains(strings.ToLower(err.Error()), "permission denied") {
                   w.WriteHeader(http.StatusServiceUnavailable)
                   _, _ = w.Write([]byte(healthzBodySocketEACCES))
                   return
               }
               slog.Warn("healthz.daemon.ping.fail", "err", err)
               w.WriteHeader(http.StatusServiceUnavailable)
               _, _ = w.Write([]byte(healthzBodyDaemonUnreach))
               return
           }

           // All green.
           w.WriteHeader(http.StatusOK)
           _, _ = w.Write([]byte(healthzBodyOK))
       }

       // getState (unchanged from Phase 1)
       func (s *Server) getState(w http.ResponseWriter, r *http.Request) {
           st := s.store.Get()
           w.Header().Set("Content-Type", "application/json; charset=utf-8")
           w.Header().Set("Cache-Control", "no-store")
           if err := json.NewEncoder(w).Encode(st); err != nil {
               slog.Error("getState: encode failed", "err", err)
               return
           }
       }

       var _ = (*state.Store)(nil)
       ```
