       import (

       // mobyClient adapts *client.Client to the internal/docker.Client
       // interface. It is safe for concurrent use; the embedded *client.Client
       // documents this contract.
       type mobyClient struct {
           c *client.Client
       }

       // NewClient constructs a mobyClient using FromEnv (so DOCKER_HOST works
       // for tests) and WithAPIVersionNegotiation (so we don't pin a
       // specific Engine API version — the daemon and client agree at first
       // call).
       //
       // The construction failure modes are wrapped with the "docker.NewClient"
       // prefix so callers can grep for the boot-time failure surface.
       func NewClient(ctx context.Context) (*mobyClient, error) {
           c, err := client.NewClientWithOpts(client.FromEnv, client.WithAPIVersionNegotiation())
           if err != nil {
               return nil, fmt.Errorf("docker.NewClient: %w", err)
           }
           return &mobyClient{c: c}, nil
       }

       // Each method below delegates to the embedded client. The wrapping is
       // structural so tests can substitute a fake docker.Client without
       // touching the daemon (see moby_test.go and api/handlers_healthz_test.go
       // in plan 02-04).
       func (m *mobyClient) Ping(ctx context.Context) error {
           // The exact SDK shape is client.Client.Ping(ctx, opts) returning
           // (PingResult, error). We discard PingResult — we only care about
           // reachability. If the SDK in v0.4.1 has changed signatures
           // again, adapt and document.
           _, err := m.c.Ping(ctx, client.PingOptions{})
           return err
       }

       func (m *mobyClient) ContainerList(ctx context.Context, opts ContainerListOptions) ([]ContainerSummary, error) {
           // If the SDK in v0.4.1 returns a wrapper type rather than a slice,
           // unwrap here and document inline.
           res, err := m.c.ContainerList(ctx, opts)
           if err != nil {

           }
           // Adapt res shape to []ContainerSummary if needed. The actual line
           // depends on the SDK return type discovered via `go doc`.
           return res /* or res.Containers */, nil
       }

       func (m *mobyClient) ContainerInspect(ctx context.Context, id string) (ContainerInspect, error) {
           res, err := m.c.ContainerInspect(ctx, id, client.ContainerInspectOptions{})
           if err != nil {
               return ContainerInspect{}, fmt.Errorf("docker.ContainerInspect %s: %w", id, err)
           }
           return res, nil
       }

       func (m *mobyClient) Events(ctx context.Context, opts EventsListOptions) (<-chan EventMessage, <-chan error) {
           // If the SDK exposes Events as an iterator instead of channels,
           // adapt here: start a goroutine that drains the iterator into the

           eventCh := make(chan EventMessage, 16) // small buffer absorbs bursts
           errCh := make(chan error, 1)
           // ... iterator adapter goroutine ...

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Process → Docker daemon socket | The `mobyClient` makes Unix-socket RPC calls to a privileged daemon. Anything reachable from the daemon (root on host, all containers) is reachable from this code path. |
| Go module supply chain | `github.com/moby/moby/client@v0.4.1` and its transitive deps land in go.sum. A poisoned dependency or a typosquat (`github.com/docker/docker` lookalike) would compromise the binary. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01-01 | Tampering | go.mod / go.sum | mitigate | Pin moby/moby/client to exact `v0.4.1` (not `^0.4` or `latest`); CI re-runs `go mod verify` (handled in plan 02-04 wiring task). |
