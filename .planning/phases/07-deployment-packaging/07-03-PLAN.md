    Key implementation points (full body left to the executor — this is a SPEC, not a verbatim copy job):

    ```ts
    /**
     * deploy-portability.spec.ts — DEPLOY-05 / Phase 7 Success Criterion #1
     *
     * RED-first per C4: this spec runs against docker-compose.example.yml,
     * which is the brief §F7 reference deployment. Acceptance is:
     *   "copying docker-compose.yml to a second clean Debian 12 host and
     *    running `docker compose up -d` (with the documented user:
     *    '65532:<docker-gid>' step) produces a working install with the table
     *    loading at :8080 and no manual UI steps."
     *
     * Strategy (Phase 7 RESEARCH §7): run on the ubuntu-24.04 CI runner as a
     * surrogate Debian 12 environment. Real-Debian-12 verification is the
     * manual smoke step (Phase 7 success criterion #5).
     *
     * Gating: this spec runs only when DEPLOY_PORTABILITY=1 is in the env.
     * The default `make e2e` invocation does NOT set it, so the existing
     * Phase 1+2+3+4 suites are untouched.
     *
     * Wall-clock budget: ~30s.
     */
    import { test, expect } from "@playwright/test";
    import { execSync } from "child_process";
    import * as fs from "fs";
    import * as path from "path";
    import * as os from "os";

    const shouldRun = process.env.DEPLOY_PORTABILITY === "1";

    test.describe("DEPLOY-05 portability", () => {
      test.skip(!shouldRun, "set DEPLOY_PORTABILITY=1 to run");

      test("clean-dir compose-up produces healthz=200 + table renders", async ({ request }) => {
        const tmp = fs.mkdtempSync(path.join(os.tmpdir(), "hmi-portability-"));
        const composeOut = path.join(tmp, "docker-compose.yml");
        const stateOut = path.join(tmp, "hmi_update_state.json");

        try {
          // 1. Build a local "portability" tag from the production Dockerfile.
          //    Plan 07-01's image is built fresh; we trust the cache.
          execSync("docker build -t hmi-update:portability .", {
            cwd: path.join(__dirname, "..", ".."),
            stdio: "inherit",
          });

          // 2. Detect the in-container docker GID via the same alpine-stat
          //    technique the Makefile uses (Pitfall 9 — macOS host GID is
          //    wrong; alpine-stat reads the in-VM socket GID which is what
          //    matters).
          const dockerGid = execSync(
            "docker run --rm -v /var/run/docker.sock:/var/run/docker.sock --entrypoint stat alpine -c '%g' /var/run/docker.sock",
            { encoding: "utf-8" }
          ).trim();
          expect(dockerGid).toMatch(/^\d+$/);

          // 3. Read docker-compose.example.yml and substitute:
          //    - ghcr.io/centroid-is/hmi-update:latest → hmi-update:portability
          //    - <docker-gid> → ${dockerGid}
          //    - 8080:8080 → 8081:8080
          //    - /opt/centroid/docker-compose.yml → ${composeOut} (self-reference
          //      so the bind-mount points back at the file we just wrote)
          //    - /opt/centroid/hmi_update_state.json → ${stateOut}
          let compose = fs.readFileSync(
            path.join(__dirname, "..", "..", "docker-compose.example.yml"),
            "utf-8"
          );
          compose = compose
            .replace("ghcr.io/centroid-is/hmi-update:latest", "hmi-update:portability")
            .replace("<docker-gid>", dockerGid)
            .replace('"8080:8080"', '"8081:8080"')
            .replace(
              "/opt/centroid/docker-compose.yml:/host/docker-compose.yml:ro",
              `${composeOut}:/host/docker-compose.yml:ro`
            )
            .replace(
              "/opt/centroid/hmi_update_state.json:/state/hmi_update_state.json",
              `${stateOut}:/state/hmi_update_state.json`
            );
          fs.writeFileSync(composeOut, compose);

          // 4. Create the state file with the right ownership (mock the
          //    operator's chown step; in CI we already run as a user that
          //    has docker group access).
          fs.writeFileSync(stateOut, "");
          // chown 65532:65532 — fails silently on systems where the user
          // doesn't have CAP_CHOWN; CI runs as root so this works.
          try { fs.chownSync(stateOut, 65532, 65532); } catch (e) { /* dev env */ }

          // 5. Bring up the stack
          execSync(`docker compose -f ${composeOut} up -d --wait --timeout 60`, {
            stdio: "inherit",
          });

          // 6. Poll /healthz
          let healthOK = false;
          for (let i = 0; i < 30; i++) {
            try {
              const resp = await request.get("http://localhost:8081/healthz");
              if (resp.status() === 200) { healthOK = true; break; }
            } catch (e) { /* not ready yet */ }
            await new Promise((r) => setTimeout(r, 1000));
          }
          expect(healthOK).toBe(true);

          // 7. Assert the table page loads (200 + contains a known DOM hook)
          const page = await request.get("http://localhost:8081/");
          expect(page.status()).toBe(200);
          const html = await page.text();
          // The Svelte shell's title or root mount-point is stable across
          // phases. Adjust selector if Phase 5's UI lands with different
          // markup — this is the loosest possible assertion that the
          // embedded dist is being served (not the index-fallback for /api/*).
          expect(html.toLowerCase()).toContain("hmi-update");
        } finally {
          // Teardown — best-effort, always-run.
          try { execSync(`docker compose -f ${composeOut} down -v --timeout 30`, { stdio: "inherit" }); } catch (e) { /* ignore */ }
          try { fs.rmSync(tmp, { recursive: true, force: true }); } catch (e) { /* ignore */ }
        }
      });
    });
    ```
