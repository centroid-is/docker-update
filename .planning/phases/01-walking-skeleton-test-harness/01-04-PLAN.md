    ```go
    package api

    import (
        "encoding/json"
        "net/http"

        "github.com/centroid-is/hmi-update/internal/state"
    )

    // healthz returns 200 if the state file is readable; 503 otherwise.
    // Phase 2 will add the docker-socket reachability check; Phase 1 only checks state.
    func (s *Server) healthz(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Cache-Control", "no-store")
        // The fact that state.Store.NewStore succeeded at boot is the proof
        // that the state file is readable. If we got here, we're healthy.
        // Phase 2 adds a "check the docker socket" probe here.
        w.WriteHeader(http.StatusOK)
        _, _ = w.Write([]byte(`{"status":"ok"}`))
    }

    // getState returns the in-memory state snapshot as JSON.
    // No I/O on this path - per OBS-03 (Phase 4) this must be cheap enough for 5s UI polling.
    func (s *Server) getState(w http.ResponseWriter, r *http.Request) {
        st := s.store.Get()
        // Convert state.State to api.State if the two types diverge - in Phase 1 they share field shapes.
        w.Header().Set("Content-Type", "application/json; charset=utf-8")
        w.Header().Set("Cache-Control", "no-store")
        if err := json.NewEncoder(w).Encode(st); err != nil {
            // can't write headers now; log and bail
            return
        }
    }
    ```

    Step 3: Create `internal/api/server.go`:
    ```go
    package api

    import (
        "net/http"
        "time"

        "github.com/centroid-is/hmi-update/internal/state"
    )

    type Server struct {
        store *state.Store
        mux   *http.ServeMux
    }

    func NewServer(store *state.Store) *Server {
        s := &Server{store: store, mux: http.NewServeMux()}
        s.routes()
        return s
    }

    func (s *Server) routes() {
        s.mux.HandleFunc("GET /healthz", s.healthz)
        s.mux.HandleFunc("GET /api/state", s.getState)
        // Static handler matches /, /index.html, and /assets/*
        s.mux.Handle("/", newStaticHandler())
    }

    func (s *Server) Handler() http.Handler { return s.mux }

    func (s *Server) ListenAndServe(addr string) error {
        srv := &http.Server{
            Addr:         addr,
            Handler:      s.mux,
            ReadTimeout:  10 * time.Second,  // Slow-loris mitigation per Phase 1 security domain
            WriteTimeout: 10 * time.Second,
        }
        return srv.ListenAndServe()
    }
    ```
